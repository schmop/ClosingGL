function(searchIncludePaths LIB_SRC LIB_DIR path)
    file(GLOB tmp ${path}/*)
    message("Include paths found in ${path}: ${tmp}")
    list(APPEND ${LIB_SRC} "${tmp}")
    list(APPEND ${LIB_DIR} "${path}")
    set(${LIB_SRC} ${LIB_SRC} PARENT_SCOPE)
    set(${LIB_DIR} ${LIB_DIR} PARENT_SCOPE)
    include_directories(${CMAKE_CURRENT_SOURCE_DIR}/${path})
endfunction(searchIncludePaths)
